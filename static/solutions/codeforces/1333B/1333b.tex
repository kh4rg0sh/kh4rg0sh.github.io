void solve() {
    int n;
    std::cin >> n;

    std::vector<ll> a(n + 1), b(n + 1);
    ll ones = 0, neg = 0;
    for (ll i = 1; i <= n; i++) {
        std::cin >> a[i];
        if (a[i] == 1) ones++;
        else if (a[i] == -1) neg++;
    }

    for (ll i = 1; i <= n; i++) {
        std::cin >> b[i];
    }

    bool yes = true;
    for (ll i = n; i > 0; i--) {
        if (a[i] == 1) ones--;
        else if (a[i] == -1) neg--;
        if (a[i] ==  b[i]) continue;
        if (a[i] < b[i]) {
            if (ones == 0) {
                yes = false;
                break;
            }
        } else {
            if (neg == 0) {
                yes = false;
                break;
            }
        }
    }

    if (yes) {
        std::cout << "YES\n";
    } else {
        std::cout << "NO\n";
    }
}
