\section{Problem}
\begin{problem*}[1st AGO Shortlist G0]
    Let $ABC$ be an acute triangle, and let $S$ be the intersection of the tangents from $B$ and $C$ to circumcircle of $ABC$. Let $D$ be the projection of $A$ onto line $BC$ and $M$ be the midpoint of $BC$, $N$ be the midpoint of $AS$. Prove that $ND=NM$.
\end{problem*}

\section{Solution}
\begin{figure}[h]
    \centering
    \begin{asy}
        import geometry;
        size(8cm); defaultpen(fontsize(10pt));

        pair A, B, C, O;
        O = origin;
        A = dir(121);
        B = dir(215);
        C = dir(325);

        pair BB = rotate(90, B) * O;
        pair CC = rotate(90, C) * O;

        pair S = extension(B, BB, C, CC);
        pair M = (B + C) / 2;
        pair N = (A + S) / 2;
        pair D = foot(A, B, C);
        pair MM = 2 * N - M;

        dot("$A$", A, dir(140));
        dot("$B$", B, dir(225));
        dot("$C$", C, dir(315));
        dot("$S$", S, dir(225));
        dot("$D$", D, dir(225));
        dot("$M$", M, dir(315));
        dot("$N$", N, dir(45));
        dot("$M'$", MM, dir(135));

        draw(A--B--C--cycle);
        draw(circumcircle(A, B, C));
        draw(B--S); draw(C--S);
        draw(A--S); draw(A--D); 
        draw(M--S); draw(N--S); 
        draw(M--MM); markrightangle(C, D, A, 7); draw(N--D);
        draw(arc(circumcircle(D, MM, M), -230, -10), heavygray+dashed);
    \end{asy}
\end{figure}
\begin{proof}
    Let $M'$ be the reflection of $M$ over $N$. Since $\overline{M'N}$ $=$ $\overline{NM}$ and $\overline{AN}$ $=$ $\overline{NS}$ $\implies$ $\triangle ANM'$ $\cong$ $\triangle SNM$ by SAS congruence criterion. This implies that $\overline{AM'}$ $\parallel$ $\overline{SM}$, but $\overline{SM}$ $\perp$ $\overline{BC}$ $\implies$ $\overline{AM'}$ $\perp$ $\overline{BC}$. From here we conclude that $M'$ lies on $\overline{AD}$, but $N$ is the midpoint of the hypotenuse of right-angled triangle $\triangle M'DM$ $\implies$ $N$ is the center of $\odot(M'DM)$. Thus, $\overline{ND}$ $=$ $\overline{DM}$.
\end{proof}
