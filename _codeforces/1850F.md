we just want to count the number of frogs that land for each position j <= n. this can be computed in O(nlogn) by getting number of frogs with hop length k and marking all multiples of k <= n with that. finally we get the maximum value marked among the first n values.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1), b(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        if (a[i] <= n) b[a[i]]++;
    }

    vector<i64> freq(n + 1);
    for (int i = 1; i <= n; i++) {
        for (int j = i; j <= n; j += i) {
            freq[j] += b[i];
        }
    }

    i64 best = 0;
    for (int i = 1; i <= n; i++) {
        best = max(best, freq[i]);
    }
    cout << best << '\n';
}
```
