just decompose the number into its place value decomposition and skip the digits with 0 face value.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> temp;
    i64 power = 1;
    while (n > 0) {
        i64 rem = n % 10;
        if (rem > 0) {
            temp.push_back(rem * power);
        }
        power *= 10;
        n /= 10;
    }
    cout << temp.size() << '\n';
    for (auto& u: temp) {
        cout << u << ' ';
    }
    cout << '\n';
}
```
