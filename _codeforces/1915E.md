flip the signs of all numbers on the odd position. now the problem is just asking if there exists a zero sum subarray, which is a very famous problem.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        if (i % 2) a[i] = -a[i];
    }

    map<i64, i64> mp;
    i64 sum = 0; mp[sum]++;
    bool found = false;
    for (int i = 1; i <= n; i++) {
        sum += a[i];
        if (mp.count(sum)) {
            found = true;
            break;
        }
        mp[sum]++;
    }
    cf(found);
}
```
