split the string into arrays of consecutive A's. we want to take as many A's as we can. however, we cannot take all A's if each A filled subarray is separated only by single B and the string starts and ends with A. in that case, we will have to drop the smallest length A filled subarray. otherwise we can always take all As.

```c++
void solve() {
    string s; cin >> s;
    i64 n = s.length();
    i64 l = 0, r = n - 1;
    while (l < n && s[l] == 'B') l++;
    while (r >= 0 && s[r] == 'B') r--;

    i64 count = 0, block = 0;
    vector<i64> temp;
    bool found = false;
    for (int i = l; i <= r; i++) {
        if (s[i] == 'A') {
            if (block > 0) {
                block = 0;
            }
            count++;
        } else {
            if (count > 0)
                temp.push_back(count);
            count = 0;
            block++;
            if (block > 1)
                found = true; 
        }
    }
    if (count > 0) temp.push_back(count);

    if (temp.size() == 0) {
        cout << 0 << '\n';
        return;
    }
    
    sort(temp.begin(), temp.end());
    i64 sum = 0;
    for (auto& u: temp) sum += u;

    if (found) {
        cout << sum << '\n';
    } else {
        if (l == 0 && r == n - 1) {
            cout << sum - temp[0] << '\n';
        } else {
            cout << sum << '\n';
        }
    }
}
```
