suppose L is the diameter of the tree with endpoints [u, v]. then if the tree is rooted at any node r, the value is max(d[r, u], d[r, v]). to get the loss, we compute distances from root 1. the profit is k * value - c * d[1, r].

```c++
void solve() {
    i64 n, k, c; cin >> n >> k >> c;
    vector<vector<i64>> adj(n + 1);
    for (int i = 1; i < n; i++) {
        i64 u, v; cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    Graph G(n, adj);
    auto [u, v] = G.endpoints();
    i64 best = 0;
    for (int i = 1; i <= n; i++) {
        best = max(best, max(G.d1[i], G.d2[i]) * k 
                            - c * G.d[i]);
    }
    cout << best << '\n';
}
```
