to include a number a[i], we must make sure there is no a[j] such that (a[i]a[j]) is a perfect square. which means if p[i] and p[j] are their prime factorisation masks (prime factor exponents mod 2), then their sum should be all 0. so for each number we can hash (p, e % 2) if and only if exponent is 1 mod 2 and hash this entire prime factorisation mask vector. this allows us to obtain an O(logn) equality test to check if a collision exist in the subsegment. if it exist, we start a new segment from the current index. 

```c++
void solve() {
    i64 n, k; cin >> n >> k;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];
    i64 ans = 1;
    using T = pair<i64, i64>;
    set<i64> s;
    for (int i = 1; i <= n; i++) {
        vector<T> vec;
        i64 temp = a[i];
        while (temp > 1) {
            i64 p = lp[temp];
            i64 now = 0;
            while (temp % p == 0) {
                temp /= p; now++;
            }
            if (now % 2 == 1)
                vec.push_back({p, 1});
        }
        vector_hash<T, pair_hash<i64, i64>> hasher;
        size_t h = hasher(vec);
        if (s.count(h)) {
            ans++;
            s.clear();
            s.insert(h);
        } else s.insert(h);
    }
    cout << ans << '\n';
}
```
