apply it to the minimum number.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    i64 zero = 0;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        if (a[i] == 0) zero++;
    }

    if (zero > 1) {
        cout << 0 << '\n';
    } else {
        if (zero == 1) {
            i64 prod = 1;
            for (int i = 1; i <= n; i++) {
                if (a[i] == 0) continue;
                prod *= a[i];
            }
            cout << prod << '\n';
        } else {
            sort(a.begin() + 1, a.end());
            a[1]++;
            i64 prod = 1;
            for (int i = 1; i <= n; i++) {
                prod *= a[i];
            }
            cout << prod << '\n';
        }
    }
}
```
