keep trimming both ends until both ends are unequal.

```c++
void solve() {
    i64 n; cin >> n;
    string s; cin >> s;
    deque<i64> dq;
    for (int i = 0; i < n; i++) {
        if (s[i] == '0') dq.push_back(0);
        else dq.push_back(1);
    }

    while (dq.size() > 1 && dq.back() != dq.front()) {
        dq.pop_back(); dq.pop_front();
    }

    cout << dq.size() << '\n';
}
```
