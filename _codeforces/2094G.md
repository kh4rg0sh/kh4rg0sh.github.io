the idea is to maintain both forward order and reverse order in a deque and maintain the required value and total sum under operation 1 and 3. when operation 2 occurs, we just flip reporting value from forward order to reverse order and vice versa.

```c++
void solve() {
    i64 q; cin >> q;
    deque<i64> arr, rev;
    bool turn = false;
    i64 sum = 0, val = 0, rval = 0;
    while (q--) {
        i64 s; cin >> s;
        if (s == 1) {
            if (turn) {
                i64 elem = rev.back();
                rev.pop_back();
                rev.push_front(elem);
                rval += sum - arr.size() * elem; 

                elem = arr.front();
                arr.pop_front();
                arr.push_back(elem);
                val -= sum;
                val += elem * arr.size();
            } else {
                i64 elem = arr.back();
                arr.pop_back();
                arr.push_front(elem);
                val += sum - arr.size() * elem;

                elem = rev.front();
                rev.pop_front();
                rev.push_back(elem);
                rval -= sum;
                rval += elem * arr.size();
            }
        } else if (s == 2) {
            turn = 1 - turn;
        } else {
            i64 k; cin >> k;
            sum += k;
            if (turn) {
                rev.push_back(k);
                arr.push_front(k);
                rval += k * arr.size();
                val += sum;
            } else {
                arr.push_back(k);
                rev.push_front(k);
                val += k * arr.size();
                rval += sum;
            }
        }

        if (turn) cout << rval << '\n';
        else cout << val << '\n';
    }
}
```
