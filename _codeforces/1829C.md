compute lowest for both, first and second and if first and second exist then for both min of computed and first + second. report the answer for both.

```c++
void solve() {
    i64 ans = 1e18;
    i64 n; cin >> n;
    i64 first = 1e18, second = 1e18;
    for (int i = 1; i <= n; i++) {
        i64 m; cin >> m;
        string s; cin >> s;
        if (s[0] == '1' && s[1] == '1') {
            ans = min(ans, m);
        } else if (s[0] == '1' && s[1] == '0') {
            first = min(first, m);
        } else if (s[0] == '0' && s[1] == '1') {
            second = min(second, m);
        }
    }
    if (first < 1e18 && second < 1e18) {
        ans = min(ans, first + second);
    }
    if (ans == 1e18) ans = -1;
    cout << ans << '\n';
}
```
