answer does not exist if n < 4. Put all odd numbers in descending order, then 4, 2 and then all remaining even numbers in ascending order.

```c++
void solve() {
    i64 n; cin >> n;
    if (n < 4) {
        cout << -1 << '\n';
    } else {
        stack<i64> s;
        for (int i = 1; i <= n; i += 2) s.push(i);
        while (s.size()) {
            auto f = s.top(); s.pop();
            cout << f << ' ';
        }
        cout << 4 << ' ' << 2 << ' ';
        for (int i = 6; i <= n; i += 2) {
            cout << i << ' ';
        }
        cout << '\n';
    }
}   
```
