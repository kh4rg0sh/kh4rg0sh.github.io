if there are atleast one even and odd number, we can entirely sort the array. otherwise we cannot do anything.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);

    i64 even = 0, odd = 0;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        if (a[i] % 2) odd++;
        else even++;
    }
    if (even == 0 || odd == 0) {
        for (auto& u: a) {
            if (u == 0) continue;
            cout << u << ' ';
        }
        cout << '\n';
    } else {
        sort(a.begin() + 1, a.end());
        for (auto& u: a) {
            if (u == 0) continue;
            cout << u << ' ';
        }
        cout << '\n';
    }
}
```
