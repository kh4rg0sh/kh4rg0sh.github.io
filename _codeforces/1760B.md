take the maximum over all characters in the string.

```c++
void solve() {
    i64 n; cin >> n;
    string s; cin >> s;
    i64 best = 0;
    for (auto& u: s) {
        best = max(best, (i64) u - 'a' + 1);
    }
    cout << best << '\n';
}
```
