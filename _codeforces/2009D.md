for every pair of points with same x[i], we need to add (n - 2) to the answer. and for every (x[i], y[i]), we need to check if there is an isosceles right angle triangle, right angled at this point, and add one for it to the answer.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> x(n + 1), y(n + 1);
    map<i64, i64> mp; using T = array<i64, 2>;
    map<T, bool> exist;
    for (int i = 1; i <= n; i++) {
        cin >> x[i] >> y[i];
        mp[x[i]]++; exist[{x[i], y[i]}] = true;
    }

    i64 ans = 0;
    for (auto& [u, v]: mp) {
        if (v == 2) ans += n - 2;
    }
    for (int i = 1; i <= n; i++) {
        if (exist.count({x[i] - 1, 1 - y[i]}) 
            && exist.count({x[i] + 1, 1 - y[i]})) {
            ans++;
        }
    }
    cout << ans << '\n';
}
```
