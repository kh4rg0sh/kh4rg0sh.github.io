use binary search to find what segment does d lie in and compute the speed of car in that segment and get the time.

```c++
void solve() {
    i64 n, k, q; cin >> n >> k >> q;
    vector<i64> a(k + 1), b(k + 1);
    for (int i = 1; i <= k; i++) cin >> a[i];
    for (int i = 1; i <= k; i++) cin >> b[i];

    while (q--) {
        i64 d; cin >> d;
        i64 l = 0, r = k;
        while (l < r) {
            i64 mid = (l + r + 1) >> 1;
            if (a[mid] <= d) {
                l = mid;
            } else {
                r = mid - 1;
            }
        }
        i64 idx = l;
        i64 rem = (d - a[idx]);
        if (rem == 0) {
            cout << b[idx] << ' ';
            continue;
        }

        i64 time = b[idx] + (rem * (b[idx + 1] - b[idx])) / (a[idx + 1] - a[idx]);
        cout << time << ' ';
    }
    cout << '\n';
} 
```
