make triplets of two largest number and one smallest number.

```c++
void solve() {
    i64 n; cin >> n;
    n *= 3;
    deque<i64> dq;
    for (int i = 1; i <= n; i++) {
        dq.push_back(i);
    }

    while (dq.size() > 0) {
        i64 x = dq.back(); dq.pop_back();
        i64 y = dq.back(); dq.pop_back();
        i64 z = dq.front(); dq.pop_front();
        cout << x << ' ' << y << ' ' << z << ' ';
    }
    cout << '\n';
}
```
