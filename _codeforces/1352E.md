just brute each subarray sum and add the indices to the answer.

```c++
void solve() {
    int n; cin >> n;
    vector<int> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    int ans = 0;
    vector<int> freq(8001);
    vector<bool> used(8001, false);
    for (int i = 1; i <= n; i++) {
        freq[a[i]]++;
    }

    for (int i = 1; i <= n; i++) {
        int sum = a[i];
        for (int j = i + 1; j <= n; j++) {
            sum += a[j];
            if (sum > 8000) break;
            if (!used[sum]) {
                ans += freq[sum];
                used[sum] = true;
            }
        }
    }

    cout << ans << '\n';
}
```
