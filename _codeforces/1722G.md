pick the first n - 2 numbers, suppose their xor is not 0, then we can set the other two numbers are x = 2^31 - 1 and (x ^ acc) as the other two numbers. suppose acc is 0, then we can pick n - 2 numbers from 1 to n - 2 instead of 0 to n - 3. and the approach still works.

```c++
void solve() {
    i64 n; cin >> n;
    i64 ans = 0;
    for (int i = 0; i < n - 2; i++) {
        ans ^= i;
    }

    if (ans > 0) {
        i64 x = (1LL << 31) - 1;
        i64 y = ans ^ x;
        for (int i = 0; i < n - 2; i++) {
            cout << i << ' ';
        }
        cout << x << ' ' << y << '\n';
    } else {
        ans = 0;
        for (int i = 1; i <= n - 2; i++) {
            ans ^= i;
        }
        
        i64 x = (1LL << 31) - 1;
        i64 y = ans ^ x;
        for (int i = 1; i <= n - 2; i++) {
            cout << i << ' ';
        }
        cout << x << ' ' << y << '\n';
    }
}
```
