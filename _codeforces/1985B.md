just brute the answer for all i <= n and take the max.

```c++
void solve() {
    i64 n; cin >> n;
    i64 best = 0, at = 0;
    for (int i = 2; i <= n; i++) {
        i64 num = (n / i);
        i64 cur = (num * (num + 1)) / 2 * i;
        if (best < cur) {
            best = cur;
            at = i;
        }
    }
    cout << at << '\n';
}
```
