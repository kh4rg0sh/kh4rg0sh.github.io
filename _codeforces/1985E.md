just brute 600 values up and down near the cuberoot of k.

```c++
void solve() {
    i64 x, y, z, k; cin >> x >> y >> z >> k;
    // if a, b, c are lengths
    // ans = (x - a + 1) * (y - b + 1) * (z - c + 1)
    // <= (x + y + z - (a + b + c) + 3)^3 / 27
    // a + b + c is minimised at a = b = c; a >= k ^{1/3}
    i64 val = icbrt(k);
    i64 ans = 0;
    vector<i64> temp; temp.push_back(x);
    temp.push_back(y); temp.push_back(z);
    sort(temp.begin(), temp.end());
    const i64 const_val = 600;
    for (int i = -const_val; i <= const_val; i++) {
        i64 v1 = val + i;
        if (v1 <= 0) continue;
        if (k % v1) continue;
        for (int j = -const_val; j <= const_val; j++) {
            i64 v2 = val + j;
            if (v2 <= 0) continue;
            if ((k / v1) % v2) continue;
            i64 v3 = (k / v1) / v2;
            vector<i64> cur;
            cur.push_back(v1); cur.push_back(v2);
            cur.push_back(v3); sort(cur.begin(), cur.end());
            if (temp[0] >= cur[0] 
                && temp[1] >= cur[1]
                    && temp[2] >= cur[2])
            ans = max(ans, (temp[0] - cur[0] + 1) 
                            * (temp[1] - cur[1] + 1)
                            * (temp[2] - cur[2] + 1));
        }
    }
    cout << ans << '\n';
}
```
