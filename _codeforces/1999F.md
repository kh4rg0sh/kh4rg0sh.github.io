suppose x = number of ones and y = number of zero. suppose we choose a subsequence that has i ones then the number of ways to choose such subsequence is binom{x}{i} * binom{y}{k - i}. we need to count this for i = (k - 1) / 2 + 1 to i = k.

```c++
void solve() {
    i64 n, k; cin >> n >> k;
    vector<i64> a(n + 1);

    i64 zero = 0, one = 0;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        if (a[i] == 0) zero++;
        else one++;
    }

    auto coeff = [&] (i64 x, i64 y) -> z1 {
        if (x < y || y < 0) return 0;
        z1 ans = fact[x] * invfact[y] * invfact[x - y];
        return ans;
    };
    
    z1 sum = 0;
    for (int i = (k - 1) / 2 + 1; i <= k; i++) {
        sum += coeff(one, i) * coeff(zero, k - i);
    }
    cout << sum << '\n';
}   
```
