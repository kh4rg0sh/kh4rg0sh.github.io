to compute value for each pair of string, for each character add up the absolute difference between the characters.

```c++
void solve() {
    i64 n, m; cin >> n >> m;
    string s[n];
    for (int i = 0; i < n; i++) {
        cin >> s[i];
    }

    auto compute = [&] (string& s, string& t) -> i64 {
        i64 cur = 0;
        for (int i = 0; i < s.length(); i++) {
            cur += abs(s[i] - t[i]);
        }
        return cur;
    };

    i64 best = 1e18;
    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            i64 ans = compute(s[i], s[j]);
            best = min(best, ans);
        }
    }
    cout << best << '\n';
}
```
