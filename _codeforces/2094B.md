just clamp the interval to where you can delete elements safely.

```c++
void solve() {
    i64 n, m, l, r; cin >> n >> m >> l >> r;
    i64 clamp = n - m;
    i64 left = 0 - l;
    i64 right = r - 0;
    if (clamp <= left) {
        cout << clamp + l << ' ' << r << '\n';
    } else {
        i64 rem = r - (clamp - left);
        cout << 0 << ' ' << rem << '\n';
    }
}
```
