void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1), b(n + 1), pref(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];
    for (int i = 1; i <= n; i++) cin >> b[i];
    for (int i = 1; i <= n; i++) pref[i] = pref[i - 1] + a[i];

    using T = array<i64, 2>;
    vector<vector<T>> adj(n + 1);
    for (int i = 1; i <= n; i++) {
        if (i > 1) adj[i].push_back({0, i - 1});
        if (b[i] > i) adj[i].push_back({a[i], b[i]});
    }

    vector<i64> d(n + 1, 1e18);
    d[1] = 0;

    priority_queue<T, vector<T>, greater<T>> pq;
    pq.push({d[1], 1});
    while (pq.size()) {
        auto [D, f] = pq.top(); pq.pop();
        if (D != d[f]) continue;
        for (auto& [ln, u]: adj[f]) {
            if (d[f] + ln < d[u]) {
                d[u] = d[f] + ln;
                pq.push({d[u], u});
            }
        }
    }

    i64 ans = 0;
    for (int i = 1; i <= n; i++) {
        ans = max(ans, pref[i] - d[i]);
    }

    cout << ans << '\n';
}
