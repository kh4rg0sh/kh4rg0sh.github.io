any number x can only be paired with (2^31 - 1) - x. we make such pairs if possible otherwise keep x as an individual number in the group. the answer is number of such pairs.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    multiset<i64> m;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        m.insert(a[i]);
    }

    i64 ans = 0;
    while (m.size()) {
        auto f = *m.begin();
        i64 ff = (1LL << 31) - 1 - f;
        if (m.count(ff)) {
            m.erase(m.find(ff));
            m.erase(m.find(f));
        } else {
            m.erase(m.find(f));
        }
        ans++;
    }
    cout << ans << '\n';
}
```
