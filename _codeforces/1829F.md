void solve() {
    i64 n, m; cin >> n >> m;
    vector<vector<i64>> adj(n + 1);
    for (int i = 1; i <= m; i++) {
        i64 x, y; cin >> x >> y;
        adj[x].push_back(y);
        adj[y].push_back(x);
    }

    // only one vertex with degree x
    // x vertices with degree y + 1
    // x*y vertices with degree 1
    map<i64, i64> mp;
    for (int i = 1; i <= n; i++) {
        mp[adj[i].size()]++;
    }

    // if x == y => only 2
    // if x == y == 1 => only 1
    // if all diff then
    i64 xy = 0, x = 0, y = 0;
    if (mp.size() == 2) {
        for (auto& [u, v]: mp) {
            if (u == 1) {
                xy = v;
            } else {
                x = u;
            }
        }
        y = x - 1;
        cout << x << ' ' << y << '\n';
    } else {
        for (auto& [u, v]: mp) {
            if (u == 1) {
                xy = 1;
            } else {
                if (v == 1) {
                    x = u;
                } else {
                    y = u - 1;
                }
            }
        }
        cout << x << ' ' << y << '\n';
    }
}
