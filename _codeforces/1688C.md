counting the frequency of each character in all the strings, only the initial character has an odd frequency. others have even.

```c++
void solve() {
    i64 n; cin >> n;
    map<char, i64> mp;
    for (int i = 1; i <= 2 * n; i++) {
        string s; cin >> s;
        for (auto& u: s) {
            mp[u]++;
        }
    }
    string s; cin >> s;
    for (auto& u: s) {
        mp[u]++;
    }

    for (auto& [u, v]: mp) {
        if (v % 2) {
            cout << u << '\n';
            return;
        }
    }
}
```
