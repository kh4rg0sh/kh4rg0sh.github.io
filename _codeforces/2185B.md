brute all possibilities and take the maximum answer.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    i64 best = 0;
    for (int i = 1; i <= n; i++) {
        for (int j = i + 1; j <= n; j++) {
            vector<i64> b = a;
            swap(b[i], b[j]);
            vector<i64> pref(n + 1);
            i64 ans = 0;
            for (int k = 1; k <= n; k++) {
                pref[k] = max(pref[k - 1], b[k]);
                ans += pref[k];
            }
            best = max(best, ans);
        }
    }

    vector<i64> b = a;
    vector<i64> pref(n + 1);
    i64 ans = 0;
    for (int k = 1; k <= n; k++) {
        pref[k] = max(pref[k - 1], b[k]);
        ans += pref[k];
    }
    best = max(best, ans);
    cout << best << '\n';
}
```
