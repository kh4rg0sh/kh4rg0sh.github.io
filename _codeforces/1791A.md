void solve() {
    char c; cin >> c;
    set<char> s;
    s.insert('c');
    s.insert('o');
    s.insert('d');
    s.insert('e');
    s.insert('f');
    s.insert('r');
    s.insert('s');
    cf(s.count(c) > 0);
}
