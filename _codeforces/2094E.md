For each k, we can compute the sum in O(30). So for all k, we can compute it in O(30n).

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);

    vector<i64> bits(31);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        for (int j = 0; j < 31; j++) {
            if ((a[i] >> j) & 1) bits[j]++;
        }
    }

    i64 best = 0;
    for (int i = 1; i <= n; i++) {
        i64 cur = 0;
        for (int j = 0; j < 31; j++) {
            if ((a[i] >> j) & 1) {
                cur += (1LL << j) * (n - bits[j]);
            } else {
                cur += (1LL << j) * bits[j];
            }
        }
        best = max(best, cur);
    }

    cout << best << '\n';
}
```
