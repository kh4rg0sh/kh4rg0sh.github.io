binary search on value of k.

```c++
void solve() {
    i64 n, c, d; cin >> n >> c >> d;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];
    sort(a.begin() + 1, a.end(), greater<i64>());

    vector<i64> pref(n + 1);
    for (int i = 1; i <= n; i++) pref[i] = pref[i - 1] + a[i];

    auto check = [&] (i64 k) -> i64 {
        // if k > n - 1 => all
        // else pref[k + 1]
        i64 sum = (k + 1 > n) ? pref[n]: pref[k + 1];
        i64 blocks = d / (k + 1);
        i64 total = blocks * sum;
        i64 rem = d % (k + 1);
        if (rem > n) rem = n;
        total += pref[rem];
        // cout << total << '\n';
        return total;
    };

    i64 M = 2e5 + 69;
    i64 l = 0, r = M;
    bool found = false;
    while (l < r) {
        // cout << l << ' ' << r << ' ';
        i64 mid = (l + r + 1) >> 1;
        if (check(mid) >= c) {
            l = mid;
            found = true;
        } else {
            r = mid - 1;
        }
    }

    if (l == M) {
        cout << "Infinity\n";
    } else {
        if (found) {
            cout << l << '\n';
        } else {
            if (c <= d * a[1]) cout << 0 << '\n';
            else cout << "Impossible\n";
        }
    }
}
```
