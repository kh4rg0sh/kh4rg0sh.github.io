start from the lowest row and move upwards and check what column does the character # lie in.

```c++
void solve() {
    i64 n; cin >> n;
    string s[n];
    for (int i = 0; i < n; i++) {
        cin >> s[i];
    }

    for (int i = n - 1; i >= 0; i--) {
        for (int j = 0; j < 4; j++) {
            if (s[i][j] == '#') cout << j + 1 << ' ';
        }
    }
    cout << '\n';
}
```
