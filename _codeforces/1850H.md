construct a directed weighted graph and compute distances using bfs and afterwards check if all the conditions are satisfied with all the computed distances.

```cpp
void solve() {
    i64 n, m; cin >> n >> m;
    using T = array<i64, 2>;
    vector<vector<T>> adj(n + 1);
    for (int i = 1; i <= m; i++) {
        i64 u, v, w; cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, -w});
    }

    bool found = true;
    vector<i64> d(n + 1, 1e18);
    for (int i = 1; i <= n; i++) {
        if (d[i] != 1e18) continue;
        d[i] = 0;

        queue<i64> q;
        q.push(i);
        while (q.size()) {
            auto f = q.front(); q.pop();
            for (auto& [v, ln]: adj[f]) {
                if (d[v] != 1e18) {
                    if (d[v] != d[f] + ln) found = false;
                    continue;
                }

                d[v] = d[f] + ln;
                q.push(v);
            }
        }
    }
    
    for (int i = 1; i <= n; i++) {
        for (auto& [u, ln]: adj[i]) {
            if (d[u] != d[i] + ln) found = false;
        }
    }

    cf(found);
}
```
