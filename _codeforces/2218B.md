compute the sum of all numbers and brute the answer for each possibility.

```c++
void solve() {
    i64 n = 7; 
    vector<i64> a(n + 1);
    i64 sum = 0;
    for (int i = 1; i <= n; i++) {
        cin >> a[i]; sum += a[i];
    }
    i64 best = -1e18;
    for (int i = 1; i <= n; i++) {
        best = max(best, 2 * a[i] - sum);
    }
    cout << best << '\n';
}
```
