sort and check if its strictly increasing.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    sort(a.begin() + 1, a.end());
    bool found = true;
    for (int i = 1; i < n; i++) {
        if (a[i + 1] <= a[i]) found = false;
    }

    cf(found);
}
```
