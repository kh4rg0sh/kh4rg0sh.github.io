we just need to check if the parities of all numbers on odd positions are same and same for even positions. if even one of them is different the answer is no.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];
    bool odd = true, even = true;
    for (int i = 3; i <= n; i++) {
        if (i % 2) {
            if ((a[i] - a[1]) % 2) odd = false;
        } else {
            if ((a[i] - a[2]) % 2) even = false;
        }
    }
    cf(odd && even);
}
```
