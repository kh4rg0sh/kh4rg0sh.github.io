since we want max over all subarrays so for the entire subarray we must have abs(n - m) <= k. if this is false we return -1 else we can begin construting. we also want the max to reach exactly k, so we can begin constructing k of the more abundant characters and after that we can alternate between 0s and 1s until one of them runs out and then just place the other after this. since abs(n - m) <= k, it guarantees that place the other after this wont exceed val == k.

```c++
void solve() {
    i64 n, m, k; cin >> n >> m >> k;
    if (abs(n - m) > k) {
        cout << -1 << '\n';
    } else {
        queue<char> zero, one;
        for (int i = 1; i <= n; i++) zero.push('0');
        for (int i = 1; i <= m; i++) one.push('1');

        string s = "";
        if (n > m) {
            for (int i = 1; i <= k; i++) {
                if (zero.size() == 0) {
                    cout << -1 << '\n';
                    return;
                }
                s += zero.front(); zero.pop();
            }
            while (one.size() && zero.size()) {
                s += one.front(); one.pop();
                s += zero.front(); zero.pop();
            }
            while (one.size()) {
                s += one.front(); one.pop();
            }
        } else {
            for (int i = 1; i <= k; i++) {
                if (one.size() == 0) {
                    cout << -1 << '\n';
                    return;
                }
                s += one.front(); one.pop();
            }
            while (one.size() && zero.size()) {
                s += zero.front(); zero.pop();
                s += one.front(); one.pop();
            }
            while (zero.size()) {
                s += zero.front(); zero.pop();
            }
        }
        cout << s << '\n';
    }
}
```
