void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    auto down = [&] (i64 x) -> i64 {
        if (x == 0) return 9;
        return x - 1;
    };

    auto up = [&] (i64 x) -> i64 {
        if (x == 9) return 0;
        return x + 1;
    };

    for (int i = 1; i <= n; i++) {
        i64 k; cin >> k;
        string s; cin >> s;
        for (int j = 0; j < k; j++) {
            if (s[j] == 'U') a[i] = down(a[i]);
            else a[i] = up(a[i]);
        }
    }
    for (int i = 1; i <= n; i++) {
        cout << a[i] << " \n"[i == n];
    }
}
