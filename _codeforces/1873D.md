just apply the operations greedily from left to right.

```c++
void solve() {
    i64 n, k; cin >> n >> k;
    i64 ans = 0;
    string s; cin >> s;
    i64 l = 0;
    while (l < n) {
        if (s[l] == 'W') {
            l++;
            continue;
        }
        ans++;
        for (int i = 1; i <= k; i++) {
            if (l + i - 1 == n) break;
            s[l + i - 1] = 'W';
        }
        l = l + k;
    }
    cout << ans << '\n';
}
```
