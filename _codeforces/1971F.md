use binary search to find a range of j for each i such that i^2 + j^2 is in the range [r, r + 1). we multiply this answer by 4 and add this answer for all i.

```c++
void solve() {
    i64 r; cin >> r;
    i64 ans = 0;
    for (i64 i = 0; i <= r; i++) {
        // j^2 + i^2 = d^2 where r <= d < r + 1
        i64 up = isqrt((r + 1) * (r + 1) - i * i - 1);
        i64 down = usqrt(r * r - i * i);
        i64 range = up - down + 1; range *= 2;
        if (down == 0) range--;
        if (i > 0) ans += 2 * range;
        else ans += range;
    }
    cout << ans << '\n';
}
```
