binary search to find the index where the function turns from < 0 to > 0 and then brute values around this inflection point which should give the minimum.
```c++
void solve() {
    i64 n, k; cin >> n >> k;
    i128 L = k, R = k + n - 1;
    i128 l = L, r = R;
    i128 sum = (r * (r + 1)) / 2 - (l * (l - 1)) / 2;
    while (l < r) {
        i128 mid = (l + r + 1) >> 1;
        auto func = [&] (i128 idx) -> bool {
            i128 left = (idx * (idx + 1)) / 2 - (L * (L - 1)) / 2;
            left = max((i128)0LL, left);
            i128 right = (R * (R + 1)) / 2 - (idx * (idx + 1)) / 2;
            right = max((i128)0LL, right);
            return left - right < 0;
        };
        if (func(mid)) {
            l = mid;
        } else {
            r = mid - 1;
        }
    }
    i128 best = sum;
    auto calc = [&] (i128 idx) -> i128 {
        i128 left = (idx * (idx + 1)) / 2 - (L * (L - 1)) / 2;
        left = max((i128)0LL, left);
        i128 right = (R * (R + 1)) / 2 - (idx * (idx + 1)) / 2;
        right = max((i128)0LL, right);
        if (left >= right) return left - right;
        return right - left;
    };

    for (i128 i = 0; i < 100; i++) {
        i128 at = l - i;
        if (at >= k && at <= n + k - 1)
            best = min(best, calc(at));
        at = l + i;
        if (at >= k && at <= n + k - 1)
            best = min(best, calc(at));
    }

    cout << best << '\n';
}
```
