if (i, j) can be connected then either (1, i) or (1, j) can be connected. hence we will try to connect all nodes with node 1 and if all are not possible then its false. the optimal choice to connect is to choose the largest value (a[i] - i * c) to connect with node 1.

```c++
void solve() {
    i64 n, c; cin >> n >> c;
    vector<i64> arr(n + 1);
    using T = array<i64, 2>;
    vector<T> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> arr[i]; a[i][1] = i;
        a[i][0] = arr[i] - c * i;
    }

    sort(a.begin() + 1, a.end(), greater<T>());
    i64 sum = arr[1]; bool found = true;
    for (int i = 1; i <= n; i++) {
        if (a[i][1] == 1) continue;
        if (a[i][0] + sum >= 0) {
            sum += arr[a[i][1]];
        } else {
            found = false;
            break;
        }
    }
    cf(found);
}
```
