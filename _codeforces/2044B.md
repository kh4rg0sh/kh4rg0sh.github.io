replace all p by q and all q by p and reverse the string.

```c++
void solve() {
    string s; cin >> s;
    for (int i = 0; i < s.length(); i++) {
        if (s[i] == 'p') s[i] = 'q';
        else if (s[i] == 'q') s[i] = 'p';
    }
    reverse(s.begin(), s.end());
    cout << s << '\n';
}
```
