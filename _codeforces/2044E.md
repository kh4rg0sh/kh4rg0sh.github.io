just count the interval of [l1, r1] for each n. to count the length of interval, clamp the interval [l1, r1] * p between [l2, r2] and scale the interval back down by p.

```c++
void solve() {
    i64 k, l1, r1, l2, r2;
    cin >> k >> l1 >> r1 >> l2 >> r2;

    i64 ans = 0, p = 1;
    for (int i = 0; i <= 30; i++) {
        // y = x * p
        // l1 * p <= y <= r1 * p
        // l2 <= y <= r2
        i64 left = max(l2, l1 * p); left = (left + p - 1) / p;
        i64 right = min(r2, r1 * p); right = right / p;
        ans += max(0LL, right - left + 1);
        p *= k;
        if (p > 1e9) break;
    }
    cout << ans << '\n';
}
```
