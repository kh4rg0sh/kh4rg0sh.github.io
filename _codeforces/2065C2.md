iterate from left to right and make value as minimum as possible such that it is still non decreasing. (use binary search to find such value in sorted array b)

```c++
void solve() {
    i64 n, m; cin >> n >> m;
    vector<i64> a(n + 1), b(m + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];
    for (int i = 1; i <= m; i++) cin >> b[i];
    sort(b.begin() + 1, b.end());
    
    bool valid = true;
    for (int i = 1; i <= n; i++) {
        if (i == 1) {
            a[i] = min(a[i], b[1] - a[i]);
        } else {
            i64 l = 1, r = m;
            while (l < r) {
                i64 mid = (l + r) >> 1;
                if (b[mid] - a[i] >= a[i - 1]) {
                    r = mid;
                } else {
                    l = mid + 1;
                }
            }
            
            if (a[i] >= a[i - 1] && b[l] - a[i] >= a[i - 1]) {
                a[i] = min(b[l] - a[i], a[i]);
            } else {
                if (b[l] - a[i] < a[i - 1] && a[i] < a[i - 1]) {
                    valid = false;
                    break;
                } else {
                    if (b[l] - a[i] >= a[i - 1]) {
                        a[i] = b[l] - a[i];
                    }
                }
            }
        }
    }
    cf(valid);
}
```
