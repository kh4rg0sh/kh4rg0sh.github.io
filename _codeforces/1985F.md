simulate each attack and keep picking the largest power for the earliest time.

```c++
void solve() {
    i64 h, n; cin >> h >> n;
    vector<i64> a(n + 1), c(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];
    for (int i = 1; i <= n; i++) cin >> c[i];

    using T = array<i64, 4>;
    priority_queue<T, vector<T>, greater<T>> pq;
    for (int i = 1; i <= n; i++) {
        pq.push({1, -a[i], c[i], i});
    }

    i64 current = 1; // current time
    while (h > 0) {
        auto [time, attack, cooldown, index] = pq.top();
        pq.pop(); attack = -attack;
        if (time > current) {
            current = time;
        }
        h -= attack;
        pq.push({current + cooldown, -attack, cooldown});
    }

    cout << current << '\n';
}
```
