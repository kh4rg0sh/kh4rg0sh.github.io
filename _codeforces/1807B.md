if even candies are more than odd candies its yes else no.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];
    i64 x = 0, y = 0;
    for (int i = 1; i <= n; i++) {
        if (a[i] % 2) {
            y += a[i];
        } else {
            x += a[i];
        }
    }
    cf(x > y);
}
```
