binary search for each endpoint

```c++
void solve() {
    i64 n, k; cin >> n >> k;
    vector<i64> a(n + 1), h(n + 1);

    i64 best = 0;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        if (a[i] <= k) best = 1;
    }
    for (int i = 1; i <= n; i++) cin >> h[i];

    vector<i64> b(n + 1);
    for (int i = 1; i < n; i++) {
        if (h[i] % h[i + 1]) b[i] = 0;
        else b[i] = 1;
    }

    vector<i64> pref(n + 1);
    for (int i = 1; i <= n; i++) pref[i] = pref[i - 1] + a[i];

    i64 l = 1;
    while (l < n) {
        i64 r = l;
        while (r < n && b[r] == 1) r++;
        i64 len = r - l + 1;
        i64 fruits = pref[r] - pref[l - 1];

        if (fruits <= k) {
            best = max(best, len);
        } else {
            for (int j = l; j <= r; j++) {
                i64 L = l, R = j - 1;
                while (L < R) {
                    i64 mid = (L + R) >> 1;
                    if (pref[j] - pref[mid - 1] > k) {
                        L = mid + 1;
                    } else {
                        R = mid;
                    }
                }

                if (pref[j] - pref[L - 1] <= k) {
                    best = max(best, j - L + 1);
                }
            }
        }
        l = r + 1;
    }
    cout << best << '\n';
}
```
