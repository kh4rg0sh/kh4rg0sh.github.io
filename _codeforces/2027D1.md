greedy vs dp. greedy fails here. the dp idea originates if we view the problem as tiling array A into intervals baesd on weights from array B with specific costs. hence, if we want to count the minimum cost to tile the first i elements from array A using tile types of first j from array B, we can binary search to find the position k such that pref[k...i] <= b[j] and dp[i][j] = min((m - j) + dp[k - 1][j], dp[i][j - 1])

```c++
void solve() {
    i64 n, m; cin >> n >> m;
    vector<i64> a(n + 1), b(m + 1);

    i64 largest = 0;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        largest = max(largest, a[i]);
    }
    for (int i = 1; i <= m; i++) cin >> b[i];
    if (largest > b[1]) {
        cout << -1 << '\n';
        return;
    }
    
    vector<i64> pref(n + 1);
    for (int i = 1; i <= n; i++) {
        pref[i] = pref[i - 1] + a[i];
    }

    i64 ans[n + 1][m + 1];
    fill(&ans[0][0], &ans[0][0] + (n + 1)*(m + 1), 1e16);
    for (int j = 0; j <= m; j++) {
        ans[0][j] = 0;
    }

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            i64 l = 1, r = i;
            if (a[i] > b[j]) {
                ans[i][j] = ans[i][j - 1];
                continue;
            }

            while (l < r) {
                i64 mid = (l + r) >> 1;
                if (pref[i] - pref[mid - 1] <= b[j]) {
                    r = mid;
                } else {
                    l = mid + 1;
                }
            }
            ans[i][j] = min((m - j) + ans[l - 1][j], 
                            ans[i][j - 1]);
        }
    }
    cout << ans[n][m] << '\n';
}
```
