void solve() {
    string s; cin >> s;
    i64 val = (s[1] - '0') + (s[0] - '0') * 10;

    if (val > 12) {
        if (val - 12 >= 10)
            cout << val - 12;
        else cout << "0" << val - 12;
        cout << ":" << s[3] << s[4] << " " << "PM\n";
    } else {
        if (val == 0) {
            cout << "12";
        } else {
            if (val >= 10) 
                cout << val;
            else cout << "0" << val;
        }
        cout << ":" << s[3] << s[4] << " ";
        if (val == 12) cout << "PM\n";
        else cout << "AM\n";
    }
}
