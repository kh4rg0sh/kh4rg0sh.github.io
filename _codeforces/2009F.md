lets divide the array into n arrays of cyclic shifts. if we can tell that the left & right boundary starts from what element and what subarray, we can answer the sum of subarray by using the precomputed prefix sums.

```c++
void solve() {
    i64 n, q; cin >> n >> q;
    vector<i64> a(n + 1), pref(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];
    for (int i = 1; i <= n; i++) pref[i] = pref[i - 1] + a[i];

    while (q--) {
        i64 l, r; cin >> l >> r;
        // l = cn + d
        i64 c = (l - 1) / n, d = (l - 1) % n; d++;
        // r = en + f;
        i64 e = (r - 1) / n, f = (r - 1) % n; f++;
        i64 sum = pref[n] * (max(0LL, e - c - 1));
        
        if (e == c) {
            i64 left = d + c; 
            if (left > n) left -= n;
            i64 right = f + c;
            if (right > n) right -= n;

            if (left <= c) {
                sum += pref[c] - pref[left - 1];
            } else {
                sum += pref[c] + (pref[n] - pref[left - 1]);
            }

            if (right <= c) {
                sum -= pref[c] - pref[right];
            } else {
                sum -= pref[c] + (pref[n] - pref[right]);
            }

        } else {
            i64 left = d + c;
            if (left > n) left -= n;
            if (left <= c) {
                sum += pref[c] - pref[left - 1];
            } else {
                sum += pref[c] + (pref[n] - pref[left - 1]);
            }

            i64 right = f + e;
            if (right > n) right -= n;
            if (right <= e) {
                sum += (pref[n] - pref[e]) + pref[right];
            } else {
                sum += pref[right] - pref[e];
            }
        }
        
        cout << sum << '\n';
    }
}
```
