we just want to count number of segments that have already appeared before processing the ith segment such that j < i && a[j] >= a[i]. since n <= 1000, therefore we can just iterate and count in O(n^2).

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    vector<i64> b(n + 1);
    i64 ans = 0;

    for (int i = 1; i <= n; i++) {
        i64 count = 0;
        for (int j = a[i]; j <= n; j++) {
            count += b[j];
        }
        ans += count;
        b[a[i]]++;
    }   
    cout << ans << '\n';
}
```
