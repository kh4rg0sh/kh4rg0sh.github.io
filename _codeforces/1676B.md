shave off all extra when compared to the minimum.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1); i64 sum = 0, m = 1e18;
    for (int i = 1; i <= n; i++) {
        cin >> a[i]; sum += a[i];
        m = min(a[i], m);
    }
    cout << sum - m * n << '\n';
}
```
