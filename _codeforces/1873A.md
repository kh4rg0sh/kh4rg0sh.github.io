brute all permutations and check if its abc.

```c++
void solve() {
    string s; cin >> s;
    if (s == "abc") {
        cout << "YES\n";
        return;
    }
    for (int i = 0; i < 3; i++) {
        for (int j = i + 1; j < 3; j++) {
            string t = s;
            swap(t[i], t[j]);
            if (t == "abc") {
                cout << "YES\n";
                return;
            }
        }
    }
    cout << "NO\n";
}
```
