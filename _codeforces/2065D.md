we just need to sort all the arrays in descending order of their total sum and concatenate them. to break ties, choose any order it doesnt afefct the total answer and this can be proven.

```c++
void solve() {
    i64 n, m; cin >> n >> m;
    vector<vector<i64>> a(n + 1, vector<i64>(m + 1));
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) cin >> a[i][j];
    }

    using T = array<i64, 2>;
    vector<T> sum(n + 1);
    for (int i = 1; i <= n; i++) {
        i64 s = 0;
        for (int j = 1; j <= m; j++) s += a[i][j];
        sum[i][0] = s;
        sum[i][1] = i;
    }

    sort(sum.begin() + 1, sum.end(), greater<T>());
    i64 ans = 0, cur = 0;
    for (int i = 1; i <= n; i++) {
        i64 idx = sum[i][1];
        for (int j = 1; j <= m; j++) {
            cur += a[idx][j];
            ans += cur;
        }
    }
    cout << ans << '\n';
}
```
