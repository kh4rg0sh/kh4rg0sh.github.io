just query (x, x). if answer is x*x, then the missing number is greater than this. else the missing number is equal or smaller than this. enough to build a binary search on this.

```c++
i64 query(i64 a, i64 b) {
    cout << "? " << a << " " << b << std::endl;
    i64 res; cin >> res; return res;
}

void solve() {
    i64 l = 2, r = 999;
    while (l < r) {
        i64 mid = (l + r) >> 1;
        if (query(mid, mid) == mid * mid) {
            l = mid + 1;
        } else {
            r = mid;
        }
    }

    cout << "! " << l << std::endl;
}
```
