just store all the frequencies in (x + y) and (x - y), x and y. and count the frequencies for each point.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> x(n + 1), y(n + 1);

    map<i64, i64> criss, cross;
    map<i64, i64> h, v;
    for (int i = 1; i <= n; i++) {
        cin >> x[i] >> y[i];
        criss[x[i] + y[i]]++;
        cross[x[i] - y[i]]++;
        h[y[i]]++; v[x[i]]++;
    }

    i64 ans = 0;
    for (int i = 1; i <= n; i++) {
        i64 a = x[i], b = y[i];
        ans += max(0LL, criss[a + b] - 1);
        ans += max(0LL, cross[a - b] - 1);
        ans += max(0LL, h[b] - 1);
        ans += max(0LL, v[a] - 1);
    }
    cout << ans << '\n';
}
```
