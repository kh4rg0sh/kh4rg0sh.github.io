for each line count the number of times 1 appears. if there are more than 1 of these counts then its a triangle or a square.

```c++
void solve() {
    i64 n; cin >> n;
    string s[n];
    for (int i = 0; i < n; i++) {
        cin >> s[i];
    }

    set<i64> S;
    for (int i = 0; i < n; i++) {
        i64 cnt = 0;
        for (int j = 0; j < n; j++) {
            if (s[i][j] == '1') cnt++;
        }
        if (cnt == 0) continue;
        S.insert(cnt);
    }

    if (S.size() == 1) {
        cout << "SQUARE\n";
    } else {
        cout << "TRIANGLE\n";
    }
}
```
