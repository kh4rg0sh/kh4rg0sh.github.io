assign the top and bottom preferences first and put the ones with no preference in remaining seats.

```c++
void solve() {
    i64 a, b, c, m; cin >> m >> a >> b >> c;
    i64 ans = 0;

    i64 top = m - min(a, m);
    ans += min(a, m);

    i64 bottom = m - min(b, m);
    ans += min(b, m);

    ans += min(top + bottom, c);
    cout << ans << '\n';
}
```
