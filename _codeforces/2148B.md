the answer is just n + m.

```c++
void solve() {
    i64 n, m; cin >> n >> m;
    i64 x, y; cin >> x >> y;
    vector<i64> a(n + 1), b(m + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];
    for (int i = 1; i <= m; i++) cin >> b[i];
    cout << n + m << '\n';
}
```
