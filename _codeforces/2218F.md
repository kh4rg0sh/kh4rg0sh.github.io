if n = x + y is even then x-- else y--. for the new pair (x, y) we will constrct groups of 1 and 2 sized subtree and attach these to the root. the condition for the answer to exist is y >= x after accounting for the root.

```c++
void solve() {
    i64 x, y; cin >> y >> x;
    i64 n = x + y;
    vector<vector<i64>> adj(n + 1);
    if (n % 2) {
        if (x > 0) x--;
        else {
            cout << "NO\n";
            return;
        }
    } else {
        if (y > 0) y--;
        else {
            cout << "NO\n";
            return;
        }
    }
    if (x >= y) {
        cout << "YES\n";
        for (int i = 1; i <= x; i++) {
            adj[1].push_back(i + 1);
        }
        for (int i = 1; i <= y; i++) {
            adj[i + 1].push_back(x + 1 + i);
        }   
        for (int i = 1; i <= n; i++) {
            if (adj[i].size() == 0) continue;
            for (auto& u: adj[i]) {
                cout << i << ' ' << u << '\n';
            }
        }
    } else {
        cout << "NO\n";
    }
}   
```
