check if the sum of all elements is a square number.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1); i64 sum = 0;
    for (int i = 1; i <= n; i++) {
        cin >> a[i]; sum += a[i];
    }
    i64 val = usqrt(sum);
    cf(val * val == sum);
}
```
