Suppose b represents the indices that we dont touch. Then these indices must be in non-decreasing order. Therefore b must represent the longest non-decreasing subsequence of array A, which means all the other indices that do not belong to b must be chosen in the operations. The final value would be the sum of all these chosen elements. Hence to minimise this value we must minimise (total sum of array c - sum of c at indices in b) which means we want to maximise the value touched by the longest non-decreasing subsequence.

suppose dp[i] = value that maximise the value touched by the longest non-decreasing subsequence ending on i.

then dp[i] = dp[j] + a[i] for j <= i and a[j] <= a[i]

the final value is sum(c) - max(dp[i] over all i)

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1), c(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];

    i64 sum = 0;
    for (int i = 1; i <= n; i++) {
        cin >> c[i]; sum += c[i];
    }

    vector<i64> dp = c;
    for (int i = 1; i <= n; i++) {
        for (int j = i - 1; j > 0; j--) {
            if (a[i] >= a[j]) {
                dp[i] = max(dp[i], dp[j] + c[i]);
            }
        }
    }

    i64 best = 0;
    for (int i = 1; i <= n; i++) best = max(best, dp[i]);
    cout << sum - best << '\n';
}
```
