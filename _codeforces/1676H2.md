this time we cannot use O(n^2) like in H1, so we use segment tree to speed up range query point up date to O(nlogn).

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    
    vector<i64> base(n + 1);
    IterativeSegtree<RangeSum> tree(n, base, RangeSum(), Node(0));
    i64 ans = 0;
    for (int i = 1; i <= n; i++) {
        ans += tree.query(a[i], n + 1);
        tree.update(a[i], 1);
    }

    cout << ans << '\n';
}
```
