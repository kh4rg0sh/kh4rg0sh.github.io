we want to count only unique number of elements and we need to minus one if the parity of answer changes when compared to n.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    set<i64> s;
    for (int i = 1; i <= n; i++) {
        cin >> a[i]; s.insert(a[i]);
    }
    i64 ans = s.size();
    if ((n - ans) % 2) ans = max(0LL, ans - 1);
    cout << ans << '\n';
}
```
