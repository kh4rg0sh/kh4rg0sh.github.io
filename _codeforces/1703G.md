1. GB => x - k + a[1] + a[2] / 2
2. BG => x + a[1] / 2 + a[2] / 2 - k
so GB is better than BG
so its always better to use good before bad

so we brute over all prefix that apply good keys and bad keys over rest.
```c++
void solve() {
    i64 n, k; cin >> n >> k;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i];

    vector<i64> pref(n + 1);
    for (int i = 1; i <= n; i++) pref[i] = pref[i - 1] + a[i];

    vector<i64> suff(n + 2);
    for (i64 i = 1; i <= n; i++) {
        i64 power = 2;
        for (i64 j = i; j <= min(i + 32, n); j++) {
            suff[i] += a[j] / power;
            power *= 2;
        }
    }
    i64 best = -1e18;
    for (int i = 0; i <= n; i++) {
        best = max(best, -i * k + pref[i] + suff[i + 1]);
    }

    cout << best << '\n';
}   
```
