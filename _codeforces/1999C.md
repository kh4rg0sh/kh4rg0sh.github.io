just calculate lengths of all intervals and check if its atleast s.

```c++
void solve() {
    i64 n, s, m; cin >> n >> s >> m;
    i64 prev = 0; bool found = false;
    for (int i = 1; i <= n; i++) {
        i64 l, r; cin >> l >> r;
        i64 length = l - prev;
        if (length >= s) found = true;
        prev = r;
    }
    i64 length = m - prev;
    if (length >= s) found = true;
    cf(found);
}
```
