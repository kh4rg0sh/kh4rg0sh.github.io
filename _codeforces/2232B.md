the answer for each index i will be always atmost the answers for the previous indices. to cmopute the answer for the current index get the prefix sum and divide it by i.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }
    
    i64 sum = 0, ans = 1e18;
    for (int i = 1; i <= n; i++) {
        sum += a[i];
        i64 cur = (sum / i);
        ans = min(ans, cur);
        cout << ans << " \n"[i == n];
    }
}
```
