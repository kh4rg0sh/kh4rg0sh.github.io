the idea is to take the highest unset bits first. so we can maintain all the numbers in a set and keep picking the maximum. after picking the maximum, we perform bitwise OR to get the current answer and compute the common bitset between the current answer and the numbers in the set and perform a xor to remove these set bits to only get the unset bits.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    using T = array<i64, 3>;
    set<T> s;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        s.insert({a[i], a[i], i});
    }

    i64 ans = 0;
    vector<i64> temp;
    while (s.size() > 0) {
        auto [f, old, at] = *s.rbegin();
        i64 save = ans;
        temp.push_back(old); ans |= f;
        set<T> t;
        for (auto& [u, v, w]: s) {
            if (w == at) continue;
            t.insert({v ^ (ans & v), v, w});
        }
        swap(s, t);
        if (ans == save) break;
    }
    for (auto& [u, v, w]: s) {
        temp.push_back(v);
    }

    for (auto& u: temp) {
        cout << u << ' ';
    }
    cout << '\n';
}
```
