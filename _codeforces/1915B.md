read each rows and identify the missing letter.

```c++
void solve() {
    string s[3];
    for (int i = 0; i < 3; i++) {
        cin >> s[i];
    }

    for (int i = 0; i < 3; i++) {
        i64 sum = 0;
        for (int j = 0; j < 3; j++) {
            if (s[i][j] == 'A') sum ^= 1;
            else if (s[i][j] == 'B') sum ^= 2;
            else if (s[i][j] == 'C') sum ^= 3;
        }
        if (sum != 0) {
            if (sum == 1) cout << "A\n";
            else if (sum == 2) cout << "B\n";
            else cout << "C\n";
            return;
        }
    }
}
```
