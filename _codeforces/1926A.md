just count the frequencies of characters in the string.

```c++
void solve() {
    string s; cin >> s;
    i64 a = 0, b = 0;
    for (auto& u: s) {
        if (u == 'A') a++;
        else b++;
    }

    if (a > b) {
        cout << "A\n";
    } else {
        cout << "B\n";
    }
}
```
