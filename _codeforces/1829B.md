simple linear scan.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    i64 best = 0, cur = 0;
    for (int i = 1; i <= n; i++) {
        if (a[i] == 0) cur++;
        else {
            best = max(best, cur);
            cur = 0;
        }
    }
    best = max(best, cur);
    cout << best << '\n';
}
```
