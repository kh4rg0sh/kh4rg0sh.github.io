max xor is achieved on many pairs. one of such pairs is choosing (n - 1) and the other number fills in the unset bits of n - 1. this achieves max xor and it can be shown that such other number always exists. 

we can find n - 1 in O(n) queries by querying ? x x y y
we can find all indices such that (n - 1) | i is max by querying ? x k y k, where k is the position of n - 1.
among these we want to find the smallest which would guarantee the xor is maximum.

```c++
char query(i64 a, i64 b, i64 c, i64 d) {
    cout << "? " << a << ' ' << b << ' ' << c << ' ' << d << std::endl;
    char res; cin >> res;
    return res;
}

void solve() {
    i64 n; cin >> n;
    i64 ans = 0;
    for (int i = 1; i < n; i++) {
        char res = query(ans, ans, i, i);
        if (res == '<') ans = i;
    }

    vector<i64> temp;
    i64 small = 0; temp.push_back(small);
    for (int i = 1; i < n; i++) {
        char res = query(small, ans, i, ans);
        if (res == '<') {
            small = i;
            temp.clear();
            temp.push_back(small);
        } else if (res == '=') {
            temp.push_back(i);
        }
    }

    i64 pos = temp[0];
    for (auto& u: temp) {
        if (u == ans) continue;
        char res = query(pos, pos, u, u);
        if (res == '>') pos = u;
    }

    cout << "! " << pos << ' ' << ans << std::endl;
}
```
