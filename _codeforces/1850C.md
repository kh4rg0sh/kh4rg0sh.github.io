find the topmost letter of the string and read all the characters below it.

```c++
void solve() {
    string s = "";
    string t[8];
    for (int i = 0; i < 8; i++) cin >> t[i];
    for (int i = 0; i < 8; i++) {
        for (int j = 0; j < 8; j++) {
            if (t[i][j] != '.') {
                i64 k = i;
                while (k < 8) {
                    if (t[k][j] == '.') break;
                    s += t[k][j];
                    k++;
                }
                cout << s << '\n';
                return;
            }
        }
    }
}
```
