sort ascending and descending and check.

```c++
void solve() {
    string s; cin >> s;
    string original = s;
    sort(s.begin(), s.end());
    string t = s;
    reverse(s.begin(), s.end());
    string u = s;
    if (original == u && original == t) {
        cout << "NO\n";
    } else {
        cout << "YES\n";
        if (original == u) {
            cout << t << '\n';
        } else {
            cout << u << '\n';
        }
    }
}
```
