the idea is to realise that there can be only 2 * n - 1 starting points. for each of these values of x, we can deterministically compute whether there is such a process or not, because at every step we must pick the next max remaining number. if we dont pick it, then x will skip past the max number remaining in the array and we will have no way to remove it.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(2 * n + 1);
    for (int i = 1; i <= 2 * n; i++) {
        cin >> a[i];
    }

    sort(a.begin() + 1, a.end());
    bool found = false;
    for (int i = 1; i < 2 * n; i++) {
        i64 val = a[2 * n];
        map<i64, i64> mp;
        vector<pair<i64, i64>> vec;
        vec.push_back({a[i], a[2 * n]});

        bool cur = true;
        for (int j = 1; j < 2 * n; j++) {
            if (j == i) continue;
            mp[a[j]]++;
        }

        i64 k = n - 1;
        while (k > 0) {
            k--;
            auto [u, v] = *mp.rbegin();
            mp[u]--;
            if (v == 1) {
                mp.erase(u);
            }
            if (mp.count(val - u)) {
                mp[val - u]--;
                if (mp[val - u] == 0) {
                    mp.erase(val - u);
                }
                vec.push_back({u, val - u});
            } else {
                cur = false;
                break;
            }
            val = u;
        }

        if (cur) {
            found = true;
            cout << "YES\n";
            cout << a[i] + a[2 * n] << '\n';
            for (auto& [u, v]: vec) {
                cout << u << ' ' << v << '\n';
            }
            return;
        }
    }
    cout << "NO\n";
}
```
