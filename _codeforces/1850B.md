pick the largest value for words of length <= 10.

```c++
void solve() {
    i64 n; cin >> n;
    using T = array<i64, 2>;
    set<T> s;
    for (int i = 1; i <= n; i++) {
        i64 x, y; cin >> x >> y;
        if (x <= 10) s.insert({y, i});
    }

    auto [c, d] = *s.rbegin();
    cout << d << '\n';
}
```
