this is an interval processing problem. we process intervals [l, r] from left to right. for each left endpoint of an interval, we dont do anything. for each right endpoint, we add the sum from [l + 1, r - 1] to the answer and set tree[l] to 1 and continue.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1), b(n + 1);
    for (int i = 1; i <= n; i++) cin >> a[i] >> b[i];
    
    vector<i64> temp;
    for (int i = 1; i <= n; i++) {
        temp.push_back(a[i]);
        temp.push_back(b[i]);
    }
    sort(temp.begin(), temp.end());
    map<i64, i64> mp;
    for (int i = 0; i < temp.size(); i++) {
        mp[temp[i]] = i + 1; // coordinate compression
    }

    vector<i64> c(2 * n + 1);
    IterativeSegtree<RangeSum> tree(2 * n, c, RangeSum(), Node(0));

    map<i64, i64> pos;
    for (int i = 1; i <= n; i++) {
        pos[a[i]] = i;
        pos[b[i]] = -i;
    }

    i64 sum = 0;
    for (int i = 0; i < temp.size(); i++) {
        i64 v = temp[i];
        if (pos[v] < 0) {
            i64 idx = -pos[v];
            i64 left = mp[a[idx]];
            tree.set(left, 1);
            sum += tree.query(left + 1, i + 1);
        }
    }
    cout << sum << '\n';
}   
```
