stupid problem. just keep track of frequency of each char in the string and break ties, based on sorting s ascending and t descending.

```c++
void solve() {
    i64 q; cin >> q;
    map<char, i64> mp1, mp2;
    i64 n1 = 1, n2 = 1;
    mp1['a'] = 1; mp2['a'] = 1;
    while (q--) {
        i64 d, k; cin >> d >> k;
        string s; cin >> s;
        if (d == 1) {
            for (auto& u: s) {
                mp1[u] += k;
            }
            n1 += k * s.size();
        } else {
            for (auto& u: s) {
                mp2[u] += k;
            }
            n2 += k * s.size();
        }
        if (mp1.size() == 0 || mp2.size() == 0) {
            if (mp1.size() == 0) {
                cout << "YES\n";
            } else cout << "NO\n";
            continue;
        }
        if (mp1.begin()->first < mp2.rbegin()->first) {
            cout << "YES\n";
        } else if (mp1.begin()->first == mp2.rbegin()->first) {
            i64 m1 = n1 - mp1.begin()->second;
            i64 m2 = n2 - mp2.begin()->second;
            if (m1 == 0 && m2 == 0) {
                if (mp1.begin()->second < mp2.rbegin()->second) {
                    cout << "YES\n";
                } else cout << "NO\n";
            } else {
                if (m1 > 0 && m2 > 0) {
                    cout << "NO\n";
                } else {
                    if (m1 == 0) {
                        if (mp1.begin()->second <= mp2.rbegin()->second) cout << "YES\n";
                        else cout << "NO\n";
                    } else {
                        cout << "NO\n";
                    }
                }
            }
        } else {
            cout << "NO\n";
        }
    }
}
```
