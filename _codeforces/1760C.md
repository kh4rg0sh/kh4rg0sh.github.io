simulate the operation of removal and maximum access using a set.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    using T = array<i64, 2>; set<T> s;
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
        s.insert({a[i], i});
    }

    for (int i = 1; i <= n; i++) {
        s.erase(s.find({a[i], i}));
        auto [val, idx] = *s.rbegin();
        cout << a[i] - val << ' ';
        s.insert({a[i], i});
    }
    cout << '\n';
}
```
