just observe if you select ith index on the current iteration then the remaining numbers turn from a[j] to a[j] ^ a[i]. so in the end we want to maximise a[i] ^ a[j] for two distinct places i and j.

```c++
void solve() {
    i64 n; cin >> n;
    vector<i64> a(n + 1);
    for (int i = 1; i <= n; i++) {
        cin >> a[i];
    }

    i64 best = 0;
    for (int i = 1; i <= n; i++) {
        for (int j = i + 1; j <= n; j++) {
            best = max(best, a[i] ^ a[j]);
        }
    }
    cout << best << '\n';
}

```
