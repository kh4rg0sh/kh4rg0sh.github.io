```c++
void solve() {
    i64 n; cin >> n;
    string s; cin >> s;
    i64 x = 0, y = 0;
    for (int i = 0; i < n; i++) {
        if (s[i] == 'U') {
            y++;
        } else if (s[i] == 'D') {
            y--;
        } else if (s[i] == 'R') {
            x++;
        } else {
            x--;
        }
        if (x == 1 && y == 1) {
            cout << "YES\n";
            return;
        }
    }
    cout << "NO\n";
}
```
