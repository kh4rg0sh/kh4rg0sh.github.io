check against both types of alternating string if all letters land on the same type of number.

```c++
void solve() {
    i64 n; cin >> n;
    string s; cin >> s;
    string t[2];
    for (int i = 0; i < n; i++) {
        if (i % 2) {
            t[0] += "0";
            t[1] += "1";
        } else {
            t[0] += "1";
            t[1] += "0";
        }
    }
    bool found = false;
    for (int turn = 0; turn < 2; turn++) {
        string u = t[turn];
        map<char, char> mp;
        bool valid = true;
        for (int i = 0; i < n; i++) {
            if (mp.count(s[i])) {
                if (mp[s[i]] != u[i]) {
                    valid = false;
                }
            } else {
                mp[s[i]] = u[i];
            }
        }
        if (valid) found = true;
    }   
    cf(found);
}
```
