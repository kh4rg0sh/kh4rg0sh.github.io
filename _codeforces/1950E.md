the answer must be a divisor of n, hence we just need to solve for each divisor of n and check if such a string k exists. for each divisor, we can check this in O(n). suppose, d is divisor and k is the required string of length d. let k = s[0..d] then we check for each consecutive partition of length d and check how many mismatches occur, on what indices and what character was instead required. the passing condition is either <= 1 mismatches in this entire string, or the mismatches are exactly (n / d) - 1 and these occur for the same index of string k and the required character in each of the case was the same. since there are about n^{1/3} divisors so we can solve this in O(n^{4/3}). 

```c++
void solve() {
    i64 n; cin >> n;
    string s; cin >> s;

    i64 ans = n;
    for (int i = 1; i * i <= n; i++) {
        if (n % i) continue;

        auto compute = [&] (i64 j) -> void {
            i64 mis = 0;
            string t = "";
            for (int k = 0; k < j; k++) {
                t += s[k];
            }

            set<i64> temp;
            set<char> store;
            for (int k = 0; k < n; k++) {
                if (s[k] != t[k % j]) {
                    mis++;
                    temp.insert(k % j);
                    store.insert(s[k]);
                }
            }

            bool valid = (mis <= 1) || (mis == (n / j) - 1 
                && temp.size() == 1 && store.size() == 1);
            if (valid) {
                ans = min(ans, j);
            }
        };

        compute(i);
        if (i * i != n) compute(n / i);
    }
    cout << ans << '\n';
}
```
